51 /* Internal helper function for BCMOLT_API_NEXT_ID. */
52 static inline uint8_t bcmolt_api_next_id_generic(uint8_t cur_id, uint8_t count, uint64_t full_mask)
53 {
54  while (cur_id < count)
55  {
56  ++cur_id;
57  if ((full_mask & (1ull << cur_id)) != 0)
58  {
59  break;
60  }
61  }
62  return cur_id;
63 }

65 /* Macro to determine the next valid ID number of a given type.
66  * NOTE: the seemingly redundant cast to int is to avoid the -Wbad-function-cast false error.
67  * \param type The name of the ID type (e.g. 'bcmolt_object_id').
68  * \param val An ID number.
69  * \return The next ID number (or xx__NUM_OF if no more are available).
70  */
71 #define BCMOLT_API_NEXT_ID(type, val) \
72  ((type)((int)bcmolt_api_next_id_generic((uint8_t)(val), (uint8_t)type ## __num_of, type ## _full_mask)))

74 /* Macro to iterate over each defined value for a given ID type (object ID, field ID, etc).
75  * \param type The name of the ID type (e.g. 'bcmolt_object_id').
76  * \param var An variable with the ID enum type.
77  */
78 #define BCMOLT_API_FOR_EACH_ID(type, var) \
79  for ((var) = type ## __begin; \
80  (var) < type ## __num_of; \
81  (var) = BCMOLT_API_NEXT_ID(type, var))
