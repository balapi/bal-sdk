356 static inline bcmos_bool bcmolt_key_is_wildcard(const uint8_t *f_ptr, size_t size)
357 {
358  size_t i;
359  for (i = 0; i < size; i++)
360  {
361  if (f_ptr[i] != 0xFF)

366  return BCMOS_TRUE;
